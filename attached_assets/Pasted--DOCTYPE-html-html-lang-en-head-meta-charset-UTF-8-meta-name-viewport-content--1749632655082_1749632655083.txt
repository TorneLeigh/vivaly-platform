  // Role toggle logic
  const seekerBtn = document.getElementById('seekerBtn');

  seekerBtn.addEventListener('click', () => {

    seekerBtn.setAttribute('aria-checked', 'true');
    seekerBtn.tabIndex = 0;

    providerBtn.classList.remove('active');
    providerBtn.setAttribute('aria-checked', 'false');
    providerBtn.tabIndex = -1;

    console.log("Selected role: Parent / Seeker");

  });

  providerBtn.addEventListener('click', () => {
    providerBtn.classList.add('active');
    providerBtn.setAttribute('aria-checked', 'true');
    providerBtn.tabIndex = 0;

    seekerBtn.classList.remove('active');
    seekerBtn.setAttribute('aria-checked', 'false');
    seekerBtn.tabIndex = -1;

    console.log("Selected role: Caregiver / Provider");
    // TODO: Insert your role-switch logic here
  });

  // Burger menu logic
  const burgerBtn = document.querySelector('.burger-btn');
  const mobileMenu = document.getElementById('mobileMenu');
  const closeBtn = mobileMenu.querySelector('.close-btn');

  burgerBtn.addEventListener('click', () => {
    const expanded = burgerBtn.getAttribute('aria-expanded') === 'true';
    burgerBtn.setAttribute('aria-expanded', String(!expanded));
    if (!expanded) {
      mobileMenu.classList.add('open');
      mobileMenu.setAttribute('aria-hidden', 'false');
      mobileMenu.querySelector('a').focus();
    } else {
      mobileMenu.classList.remove('open');
      mobileMenu.setAttribute('aria-hidden', 'true');
      burgerBtn.focus();
    }
  });

  closeBtn.addEventListener('click', () => {
    mobileMenu.classList.remove('open');
    mobileMenu.setAttribute('aria-hidden', 'true');
    burgerBtn.setAttribute('aria-expanded', 'false');
    burgerBtn.focus();
  });

  // Close mobile menu on outside click (optional)
  document.addEventListener('click', (e) => {
    if (!mobileMenu.contains(e.target) && !burgerBtn.contains(e.target)) {
      mobileMenu.classList.remove('open');
      mobileMenu.setAttribute('aria-hidden', 'true');
      burgerBtn.setAttribute('aria-expanded', 'false');
    }
  });
</script>
